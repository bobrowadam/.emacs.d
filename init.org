* Authinfo auto-backup to 1Password
:PROPERTIES:
:CUSTOM_ID: authinfo-auto-backup
:END:
After saving =~/.authinfo.gpg=, push it to 1Password via the shell helper
=authinfo-backup= defined in =~/secrets.zsh=. The shell function handles
account pinning (my.1password.com, never bradwell) and round-trip hash
verification, so Emacs just invokes it asynchronously and reports.

#+begin_src elisp
  (defun bob/authinfo-backup-to-1password ()
    "Run `authinfo-backup` in zsh to push ~/.authinfo.gpg to 1Password.
Runs asynchronously so save isn't blocked. Output goes to *authinfo-backup*."
    (let* ((buf (get-buffer-create "*authinfo-backup*"))
           (proc (start-process
                  "authinfo-backup" buf
                  "zsh" "-i" "-c"
                  "source ~/secrets.zsh && authinfo-backup")))
      (set-process-sentinel
       proc
       (lambda (p _event)
         (when (eq (process-status p) 'exit)
           (if (zerop (process-exit-status p))
               (message "authinfo: backed up to 1Password")
             (message "authinfo: backup FAILED (see *authinfo-backup*)")))))))

  (defun bob/authinfo-maybe-backup ()
    "After-save hook: backup if the current buffer is ~/.authinfo.gpg."
    (when (and buffer-file-name
               (string= (expand-file-name buffer-file-name)
                        (expand-file-name "~/.authinfo.gpg")))
      (bob/authinfo-backup-to-1password)))

  (add-hook 'after-save-hook #'bob/authinfo-maybe-backup)

** Prose completion
#+begin_src elisp
  (require 'seq)
  (require 'subr-x)

  (defvar bob/prose-word-completion-frequency-file
    "~/.pi/agent/extensions/src/vendor/wordfreq-en-25000-log.tsv")

  (defvar bob/prose-word-completion-project-buffer-limit 24)
  (defvar bob/prose-word-completion-project-buffer-max-chars 60000)
  (defvar bob/prose-word-completion-prefix-index-length 4)
  (defvar bob/prose-word-completion-candidate-limit 200)

  (defvar bob/prose-word-completion--cache nil)
  (defvar bob/prose-word-completion--cache-file nil)
  (defvar-local bob/prose-word-completion--buffer-word-cache nil)

  (defun bob/prose-word-completion--bucket-key (word)
    (let ((word (downcase word)))
      (when (>= (length word) bob/prose-word-completion-prefix-index-length)
        (substring word 0 bob/prose-word-completion-prefix-index-length))))

  (defun bob/prose-word-completion--build-word-buckets (words)
    (let ((buckets (make-hash-table :test #'equal)))
      (dolist (word words buckets)
        (when-let* ((key (bob/prose-word-completion--bucket-key word)))
          (push word (gethash key buckets))))))

  (defun bob/prose-word-completion--bucket-words (buckets prefix)
    (when-let* ((key (bob/prose-word-completion--bucket-key prefix)))
      (gethash key buckets)))

  (defun bob/prose-word-completion--build-cache (file)
    (with-temp-buffer
      (insert-file-contents file)
      (let ((frequencies (make-hash-table :test #'equal)) words
            (file-lines (split-string (buffer-string) "\n" t)))
        (dolist (line file-lines)
          (pcase-let ((`(,word ,frequency . ,_)
                       (split-string line "\t")))
            (when (and word frequency
                       (string-match-p "\\`[a-z][a-z'-]+\\'" word))
              (unless (gethash word frequencies)
                (push word words))
              (puthash word (string-to-number frequency) frequencies))))
        (setq words (sort words #'string<))
        (list :words words
              :frequencies frequencies
              :buckets (bob/prose-word-completion--build-word-buckets words)))))

  (defun bob/prose-word-completion--frequency-data ()
    "Return cached word-frequency data as (WORDS . TABLE)."
    (when-let* ((file (expand-file-name bob/prose-word-completion-frequency-file))
                (file-readable-p file))
      (if (and (equal file bob/prose-word-completion--cache-file)
               bob/prose-word-completion--cache)
          bob/prose-word-completion--cache
        (setq bob/prose-word-completion--cache-file file
              bob/prose-word-completion--cache
              (bob/prose-word-completion--build-cache file)))))

  (defun bob/prose-word-completion--bounds ()
    (let ((end (point)) start)
      (when (not (looking-at-p "[[:alpha:]'-]"))
        (save-excursion
          (skip-chars-backward "[:alpha:]'-")
          (setq start (point)))
        (when (and start
                   (>= (- end start) 4)
                   (save-excursion (goto-char start) (looking-at-p "[[:alpha:]]"))
                   (not (memq (char-before start) '(?/ ?@ ?# ?$ ?. ?:))))
          (cons start end)))))

  (defun bob/prose-word-completion--score (word prefix source-score)
    (let* ((data (bob/prose-word-completion--frequency-data))
           (frequencies (plist-get data :frequencies))
           (frequency (and frequencies (gethash (downcase word) frequencies)))
           (frequency-score (if frequency
                                (* 30 (max 0 (min 1 (/ (- frequency -13.7)
                                                       (- -2.8 -13.7)))))
                              0)))
      (+ source-score frequency-score
         (- (- (length word) (length prefix))))))

  (defun bob/prose-word-completion--add-candidate (table word prefix score)
    (let ((normalized (downcase word)))
      (when (and (not (equal normalized (downcase prefix)))
                 (equal (bob/prose-word-completion--bucket-key normalized)
                        (bob/prose-word-completion--bucket-key prefix))
                 (string-match-p "\\`[[:alpha:]][[:alpha:]'-]\\{2,\\}\\'" word))
        (let ((existing (gethash normalized table)))
          (when (or (not existing) (> score (cdr existing)))
            (puthash normalized (cons word score) table))))))

  (defun bob/prose-word-completion--words-in-buffer (&optional max-chars)
    "Return cached prose words from current buffer, scanning at most MAX-CHARS."
    (let* ((tick (buffer-chars-modified-tick))
           (limit max-chars)
           (cache bob/prose-word-completion--buffer-word-cache))
      (if (and cache
               (equal (nth 0 cache) tick)
               (equal (nth 1 cache) limit))
          (nth 2 cache)
        (let ((words nil)
              (seen (make-hash-table :test #'equal))
              (case-fold-search nil))
          (save-excursion
            (save-restriction
              (widen)
              (goto-char (point-min))
              (let ((end (if limit
                             (min (point-max) (+ (point-min) limit))
                           (point-max))))
                (while (re-search-forward "[[:alpha:]][[:alpha:]'-]\\{2,\\}" end t)
                  (let* ((word (match-string-no-properties 0))
                         (normalized (downcase word)))
                    (unless (gethash normalized seen)
                      (puthash normalized t seen)
                      (push word words)))))))
          (setq words (nreverse words))
          (setq bob/prose-word-completion--buffer-word-cache
                (list tick limit words
                      (bob/prose-word-completion--build-word-buckets words)))
          (nth 2 bob/prose-word-completion--buffer-word-cache)))))

  (defun bob/prose-word-completion--bucket-words-in-buffer (prefix &optional max-chars)
    "Return cached prose words from current buffer's PREFIX bucket."
    (bob/prose-word-completion--words-in-buffer max-chars)
    (bob/prose-word-completion--bucket-words
     (nth 3 bob/prose-word-completion--buffer-word-cache) prefix))

  (defun bob/prose-word-completion--add-words (table words prefix score)
    (dolist (word words)
      (bob/prose-word-completion--add-candidate
       table word prefix
       (bob/prose-word-completion--score word prefix score))))

  (defun bob/prose-word-completion--add-buffer (table buffer prefix score
                                                      &optional max-chars)
    (when (buffer-live-p buffer)
      (with-current-buffer buffer
        (bob/prose-word-completion--add-words
         table (bob/prose-word-completion--bucket-words-in-buffer prefix max-chars)
         prefix score))))

  (defun bob/prose-word-completion--add-text (table text prefix score)
    (with-temp-buffer
      (insert text)
      (bob/prose-word-completion--add-buffer table (current-buffer) prefix score)))

  (defun bob/prose-word-completion--project-root ()
    (when-let* ((project (project-current nil)))
      (expand-file-name (project-root project))))

  (defun bob/prose-word-completion--project-file-buffers ()
    "Return open file buffers under the current project root."
    (when-let* ((root (bob/prose-word-completion--project-root)))
      (let ((buffers nil)
            (count 0))
        (dolist (buffer (buffer-list) (nreverse buffers))
          (when (and (< count bob/prose-word-completion-project-buffer-limit)
                     (not (eq buffer (current-buffer))))
            (with-current-buffer buffer
              (when-let* ((file buffer-file-name))
                (when (and (not (file-remote-p file))
                           (file-in-directory-p (expand-file-name file) root))
                  (setq count (1+ count))
                  (push buffer buffers)))))))))

  (defun bob/prose-word-completion--add-project-buffers (table prefix score)
    (dolist (buffer (bob/prose-word-completion--project-file-buffers))
      (bob/prose-word-completion--add-buffer
       table buffer prefix score
       bob/prose-word-completion-project-buffer-max-chars)))

  (defun bob/prose-word-completion--mentat-history-text ()
    "Return Mentat prompt history when the current buffer has Mentat context."
    (when (and (boundp 'mentat-prompt-root) mentat-prompt-root
               (boundp 'mentat-prompt-kind) mentat-prompt-kind
               (fboundp 'mentat--prompt-history-for)
               (fboundp 'mentat--request-prompt-history-for))
      (string-join
       (append (mentat--prompt-history-for mentat-prompt-root mentat-prompt-kind)
               (mentat--request-prompt-history-for mentat-prompt-root
                                                   mentat-prompt-kind))
       "\n")))

  (defun bob/prose-word-completion-at-point ()
    "Ranked prose word CAPF for natural-language buffers."
    (when-let* ((bounds (bob/prose-word-completion--bounds))
                (prefix (buffer-substring-no-properties (car bounds) (cdr bounds))))
      (let ((table (make-hash-table :test #'equal)))
        (bob/prose-word-completion--add-buffer table (current-buffer) prefix 300)
        (bob/prose-word-completion--add-project-buffers table prefix 180)
        (when-let* ((history (bob/prose-word-completion--mentat-history-text)))
          (bob/prose-word-completion--add-text table history prefix 200))
        (when-let* ((data (bob/prose-word-completion--frequency-data)))
          (let ((added 0))
            (catch 'done
              (dolist (word (bob/prose-word-completion--bucket-words
                             (plist-get data :buckets) prefix))
                (setq added (1+ added))
                (bob/prose-word-completion--add-candidate
                 table word prefix
                 (bob/prose-word-completion--score word prefix 60))
                (when (>= added 200)
                  (throw 'done nil))))))
        (when-let* ((data (bob/prose-word-completion--frequency-data)))
          (when (and (not (string-suffix-p "s" prefix t))
                     (gethash (downcase prefix) (plist-get data :frequencies)))
            (bob/prose-word-completion--add-candidate table (concat prefix "s")
                                                      prefix 40)))
        (let ((candidates
               (mapcar #'car
                       (sort (hash-table-values table)
                             (lambda (a b)
                               (let ((score-diff (- (cdr a) (cdr b))))
                                 (if (/= score-diff 0)
                                     (> score-diff 0)
                                   (string< (car a) (car b)))))))))
          (when candidates
            (list (car bounds) (cdr bounds)
                  (seq-take candidates bob/prose-word-completion-candidate-limit)
                  :exclusive 'no
                  :annotation-function (lambda (_) " word")))))))

  (defun bob/prose-word-completion-setup ()
    "Use Bob's ranked prose CAPF in natural-language buffers."
    (add-hook 'completion-at-point-functions
              #'bob/prose-word-completion-at-point nil t))

  (use-package text-mode
    :ensure nil
    :hook (text-mode . bob/prose-word-completion-setup))

  (defvar ediff-buffer-A)
  (defvar ediff-buffer-B)

  (defvar ediff-ancestor-buffer)

  (use-package ediff
    :ensure nil
    :custom
    (ediff-keep-variants nil)
    (ediff-diff-options "-w")
    (ediff-ignore-similar-regions t)
    (ediff-split-window-function 'split-window-horizontally)
    (ediff-window-setup-function 'ediff-setup-windows-plain))

  (defvar cc/ediff-revision-session-p nil
    "If t then `cc/ediff-revision-actual' has been called.
  This state variable is used to insert added behavior to the overridden
  function `ediff-janitor'.")

  (defun cc/ediff-revision-from-menu (e)
    "Invoke `ediff-revision' on E with variable `buffer-file-name'."
    (interactive "e")
    (cc/ediff-revision))

  (defun cc/ediff-revision ()
    "Run Ediff on the current `buffer-file-name' provided that it is `vc-registered'.
  This function handles the interactive concerns found in `ediff-revision'.
  This function will also test if a diff should apply to the current buffer."
    (interactive)
    (when (and (bound-and-true-p buffer-file-name)
               (vc-registered (buffer-file-name)))
      (if (and (buffer-modified-p)
               (y-or-n-p (format "Buffer %s is modified.  Save buffer? "
                                 (buffer-name))))
          (save-buffer (current-buffer)))
      (message buffer-file-name)
      (cc/ediff-revision-actual))

    (cond ((not (bound-and-true-p buffer-file-name))
           (message (concat (buffer-name) " is not a file that can be diffed.")))
          ((not (vc-registered buffer-file-name))
           (message (concat buffer-file-name " is not under version control.")))))

  (defun cc/ediff-revision-actual ()
    "Invoke Ediff logic to diff the modified repo file
  to its counterpart in the current branch.
  This function handles the actual diff behavior called by `ediff-revision'."
    (let ((rev1 "")
          (rev2 ""))
      (setq cc/ediff-revision-session-p t)
      (ediff-load-version-control)
      (funcall
       (intern (format "ediff-%S-internal" ediff-version-control-package))
       rev1 rev2 nil)))

  (defun ediff-janitor (ask keep-variants)
    "Kill buffers A, B, and, possibly, C, if these buffers aren't modified.
  In merge jobs, buffer C is not deleted here, but rather according to
  `ediff-quit-merge-hook'.
  ASK non-nil means ask the user whether to keep each unmodified buffer, unless
  KEEP-VARIANTS is non-nil, in which case buffers are never killed.
  A side effect of cleaning up may be that you should be careful when comparing
  the same buffer in two separate Ediff sessions: quitting one of them might
  delete this buffer in another session as well.

  CC MODIFICATION: This method overrides the original Ediff function."
    (let ((ask (if (and (boundp 'cc/ediff-revision-session-p)
                        cc/ediff-revision-session-p)
                   nil
                 ask)))
      (ediff-dispose-of-variant-according-to-user
       ediff-buffer-A 'A ask keep-variants)
      ;; !!!: CC Note: Test global state variable `cc/ediff-revision-session-p' to
      ;; determine if the modified repo file should be kept.
      ;; Guarding in place to hopefully avoid side-effects when `ediff-janitor' is
      ;; called from other Ediff functions. Informal testing has not revealed any
      ;; side-effects but YOLO.
      (if (and (boundp 'cc/ediff-revision-session-p)
               cc/ediff-revision-session-p)
          (ediff-dispose-of-variant-according-to-user
           ;; CC Note: keep-variants argument is hard-coded to t to keep
           ;; buffer holding modified repo file around.
           ediff-buffer-B 'B t t)
        (ediff-dispose-of-variant-according-to-user
         ediff-buffer-B 'B ask keep-variants))
      (if ediff-merge-job  ; don't del buf C if merging--del ancestor buf instead
          (ediff-dispose-of-variant-according-to-user
           ediff-ancestor-buffer 'Ancestor ask keep-variants)
        (ediff-dispose-of-variant-according-to-user
         ediff-buffer-C 'C ask keep-variants))
      ;; CC Note: Reset global state variable `cc/ediff-revision-session-p'.
      (if (and (boundp 'cc/ediff-revision-session-p)
               cc/ediff-revision-session-p)
          (setq cc/ediff-revision-session-p nil))))

  (defun cc/stash-window-configuration-for-ediff ()
    "Store window configuration to register 🧊.
  Use of emoji is to avoid potential use of keyboard character to reference
  the register."
    (window-configuration-to-register ?🧊))

  (add-hook 'ediff-before-setup-hook #'cc/stash-window-configuration-for-ediff)
  ;; !!!: CC Note: Why this is not `ediff-quit-hook' I do not know. But this works

      "Return non-nil if BRANCH's unique patches are already in BASE."
      (let ((cherry (magit-git-lines "cherry" base branch)))
        (and cherry
             (seq-every-p (lambda (line)
                            (string-prefix-p "-" line))
                          cherry))))

      "Return non-nil if BRANCH is ancestry-merged or squash-merged into BASE."
      (or (magit-rev-ancestor-p branch base)
          (bob/magit-cleanup--patch-equivalent-p base branch)))

      "Return local branch names deletable after merge or squash into BASE."
      (let ((protected (append bob/magit-cleanup-protected-branches
                               (list base (magit-get-current-branch)))))
        (seq-filter (lambda (branch)
                      (and (not (member branch protected))
                           (bob/magit-cleanup--branch-deletable-p base branch)))
                    (magit-list-local-branch-names))))

    (defun bob/magit-cleanup--read-branches (prompt candidates)
      "Read branch CANDIDATES for PROMPT, defaulting to all candidates."
      (let ((crm-separator "[ \t]*,[ \t]*"))
        (or (magit-completing-read-multiple prompt candidates nil t nil nil
                                            (string-join candidates ","))
            candidates)))

    (defun bob/magit-cleanup--remote-for-base (base)
      "Return the remote associated with BASE, falling back to the first remote."
      (let ((remote (magit-get "branch" base "remote")))
        (or (and remote (not (string= remote ".")) remote)
            (car (magit-list-remotes)))))

    (defun bob/magit-cleanup--remote-candidates (base local-branches)
      "Return remote branches matching LOCAL-BRANCHES deletable after merge/squash."
      (when-let* ((remote (bob/magit-cleanup--remote-for-base base))
                  (remote-base (format "%s/%s" remote base))
                  (_ (magit-rev-name remote-base)))
        (let* ((prefix (concat remote "/"))
               (protected (append bob/magit-cleanup-protected-branches (list base)))
               (remote-branches (magit-git-lines "branch" "-r" "--format=%(refname:short)")))
          (seq-filter (lambda (branch)
                        (let ((remote-branch (concat prefix branch)))
                          (and (member branch local-branches)
                               (not (member branch protected))
                               (bob/magit-cleanup--branch-deletable-p remote-base remote-branch))))
                      (delq nil
                            (mapcar (lambda (branch)
                                      (when (string-prefix-p prefix branch)
                                        (substring branch (length prefix))))
                                    remote-branches))))))

    (defun bob/magit-cleanup--maybe-delete-origin-branches (base local-branches)
      "Offer to delete matching remote LOCAL-BRANCHES merged into remote BASE."
      (when-let* ((remote (bob/magit-cleanup--remote-for-base base))
                  (candidates (bob/magit-cleanup--remote-candidates base local-branches)))
        (when (and candidates
                   (yes-or-no-p (format "Also delete %d matching merged branch(es) from %s? "
                                        (length candidates) remote)))
          (let ((selected (bob/magit-cleanup--read-branches "Delete remote branches: " candidates)))
            (when selected
              (apply #'magit-run-git-async "push" remote "--delete" selected))))))

    (defun bob/magit-cleanup-merged-branches (&optional base)
      "Delete local branches merged or squash-merged into BASE using Magit.
  Also offers to delete matching branches from the base branch's remote."
      (interactive)
      (let* ((base (or base (bob/magit-cleanup--read-base-branch)))
             (candidates (bob/magit-cleanup--local-candidates base)))
        (if (null candidates)
            (message "No local branches merged into %s" base)
          (let ((selected (bob/magit-cleanup--read-branches "Delete local merged/squashed branches: " candidates)))
            (when (and selected
                       (yes-or-no-p (format "Delete %d local branch(es) merged/squashed into %s? "
                                            (length selected) base)))
              (apply #'magit-run-git-async "branch" "-d" selected)
              (bob/magit-cleanup--maybe-delete-origin-branches base selected))))))

    (defun bob/magit-fetch-and-rebase ()
      "Fetch and rebase current branch onto main/master."
      (interactive)
      (let* ((main-branch (cond
                           ((magit-rev-name "develop") "develop")
                           ((magit-rev-name "main") "main")
                           (t "master")))
             (remote-name (or (magit-get "branch" main-branch "remote")
                              (car (magit-list-remotes))))
             (remote-branch (format "%s/%s" remote-name main-branch)))
        (message "Fetching and rebasing onto %s" remote-branch)
        (magit-run-git "fetch" remote-name main-branch)
        (magit-rebase-branch remote-branch nil)
        (when-let* ((worktree-dir (get-main-worktree main-branch)))
          (let ((default-directory worktree-dir))
            (magit-run-git-async "pull" remote-name main-branch)
            (message "Pulling worktree at %s in background..." worktree-dir)))))

    (if-let* ((root (bob/biome-root))
              (biome (expand-file-name "node_modules/.bin/biome" root))
              ((file-executable-p biome))
              (source (current-buffer))
              (file (buffer-file-name source)))
        (let* ((default-directory root)
               (extension (file-name-extension file t))
               (temp-file (make-temp-file
                           (expand-file-name
                            (format ".flymake-biome-%s-" (file-name-base file))
                            (file-name-directory file))
                           nil extension))
               (stdout (generate-new-buffer " *bob-biome-flymake*"))
               (stderr (generate-new-buffer " *bob-biome-flymake-stderr*")))
          (write-region (point-min) (point-max) temp-file nil 'silent)
          (let ((proc (make-process
                       :name "bob-biome-flymake"
                       :buffer stdout
                       :stderr stderr
                       :noquery t
                       :connection-type 'pipe
                       :command (list biome "check" "--reporter=json" temp-file))))
            (setq-local bob/biome-flymake-process proc)
            (set-process-sentinel
             proc
             (lambda (process _event)
               (when (memq (process-status process) '(exit signal))
                 (unwind-protect
                     (when (buffer-live-p source)
                       (with-current-buffer source
                         (when (eq process bob/biome-flymake-process)
                           (setq-local bob/biome-flymake-process nil)
                           (let (diags)
                             (condition-case nil
                                 (let* ((json-object-type 'plist)
                                        (json-array-type 'list)
                                        (json-false nil)
                                        (payload (with-current-buffer stdout
                                                   (goto-char (point-min))
                                                   (json-read)))
                                        (items (plist-get payload :diagnostics)))
                                   (dolist (item items)
                                     (when-let* ((loc (plist-get item :location))
                                                 (start (plist-get loc :start))
                                                 (finish (plist-get loc :end))
                                                 (line (plist-get start :line))
                                                 (column (plist-get start :column))
                                                 (end-line (plist-get finish :line))
                                                 (end-column (plist-get finish :column)))
                                       (push (flymake-make-diagnostic
                                              source
                                              (bob/biome-flymake--pos line column)
                                              (bob/biome-flymake--pos end-line end-column)
                                              (bob/biome-flymake--type
                                               (plist-get item :severity))
                                              (list "biome"
                                                    (plist-get item :category)
                                                    (plist-get item :message)))
                                             diags))))
                               (error nil))
                             (funcall report-fn (nreverse diags))))))
                   (when (file-exists-p temp-file)
                     (delete-file temp-file))
                   (when (buffer-live-p stdout)
                     (kill-buffer stdout))
                   (when (buffer-live-p stderr)
                     (kill-buffer stderr))))))))
      (funcall report-fn nil)))

* ejc-sql — JDBC-backed SQL client

table/column names pulled live from the DB, ElDoc for tables, pretty result
buffers. Requires =leiningen= on =$PATH= and Java (already installed).

=~/.emacs.d/jdbc/postgresql.jar=.

password from SSM, start an SSM tunnel, and then connect via per-environment

connections:

* linear-emacs
#+begin_src elisp
  (use-package linear-emacs
    :ensure (:repo "bobrowadam/linear-emacs" :fetcher github :files ("*.el")
                   :remotes (("upstream" :repo "anegg0/linear-emacs")))
    :demand t
    :after (org-agenda)
    :custom
    (linear-emacs-api-key (auth-source-pick-first-password
                           :host "linear.app"
                           :user "linear_api_key"))
    (linear-emacs-issues-state-mapping '(("Todo" . "TODO")
                                         ("In Progress" . "NEXT")
                                         ("In progress" . "NEXT")
                                         ("Started" . "NEXT")
                                         ("Doing" . "NEXT")
                                         ("In Review" . "WAITING")
                                         ("Blocked" . "WAITING")
                                         ("Backlog" . "BACKLOG")
                                         ("Later" . "BACKLOG")
                                         ("Working in production" . "DONE")))
    (linear-emacs-org-file-path (format "%s/%s" org-directory "linear-emacs.org"))
    :config
    (setenv "LINEAR_API_KEY" (auth-source-pick-first-password
                              :host "linear.app"
                              :user "linear_api_key"))
    (defvar bob/linear-refresh-in-progress nil)
    (defvar bob/linear-suppress-refresh nil)

    (defun bob/linear-refresh-before-org-agenda (&rest _)
      (unless (or bob/linear-refresh-in-progress
                  bob/linear-suppress-refresh)
        (setq bob/linear-refresh-in-progress t)
        (linear-emacs-list-issues)))

    (defun bob/linear-schedule-unscheduled-next ()
      "Schedule unscheduled NEXT items in `linear-emacs-org-file-path' for today."
      (when (file-exists-p linear-emacs-org-file-path)
        ;; `linear-emacs-list-issues' writes the file from a temp buffer, so an
        ;; existing visiting buffer can be stale when agenda refreshes with `g'.
        (when-let* ((buffer (get-file-buffer linear-emacs-org-file-path)))
          (with-current-buffer buffer
            (revert-buffer :ignore-auto :noconfirm)))
        (with-current-buffer (find-file-noselect linear-emacs-org-file-path)
          (org-with-wide-buffer
           (org-map-entries
            (lambda ()
              (when (and (string= (org-get-todo-state) "NEXT")
                         (not (org-entry-get nil "SCHEDULED")))
                (org-schedule nil (format-time-string "%Y-%m-%d"))))
            "/NEXT" 'file))
          (save-buffer))))

    (defun bob/linear-finish-refresh-when-idle ()
      ;; `linear-emacs-list-issues' is async; its function returns before the
      ;; file is written.  Wait until the Linear request callback has completed.
      (if (and (boundp 'linear-emacs--active-requests)
               (> linear-emacs--active-requests 0))
          (run-at-time 0.5 nil #'bob/linear-finish-refresh-when-idle)
        (bob/linear-schedule-unscheduled-next)
        (setq bob/linear-refresh-in-progress nil)
        (when-let* ((win (get-buffer-window "*Org Agenda*" t)))
          (with-selected-window win
            (let ((bob/linear-suppress-refresh t))
              (org-agenda-redo-all))))))

    (defun bob/linear-post-refresh-agenda (&rest _)
      (run-at-time 0.5 nil #'bob/linear-finish-refresh-when-idle))

    (advice-mapc (lambda (advice _props)
                   (when (string-match-p "linear-emacs-list-issues"
                                         (prin1-to-string advice))
                     (advice-remove 'org-agenda advice)))
                 'org-agenda)
    (advice-mapc (lambda (advice _props)
                   (advice-remove 'linear-emacs-list-issues advice))
                 'linear-emacs-list-issues)
    (advice-mapc (lambda (advice _props)
                   (when (string-match-p "linear-emacs-list-issues"
                                         (prin1-to-string advice))
                     (advice-remove 'org-agenda-redo advice)))
                 'org-agenda-redo)
    (advice-remove 'org-agenda-redo 'bob/linear-refresh-before-org-agenda)
    (advice-add 'org-agenda :before #'bob/linear-refresh-before-org-agenda)
    (advice-add 'linear-emacs-list-issues :after #'bob/linear-post-refresh-agenda))
